def check_controller?
  case self.the_controller
    when 'articles'
      if self.the_action == 'index'
        errors.add(:the_action, '-> Darf nicht -index- sein')
      end
      return true
    when 'news'
      return false
    when ''
      self.the_action = ''
      self.the_id = nil
      return false
    else
      self.the_action = 'index'
      self.the_id = nil
      return false
  end
end
